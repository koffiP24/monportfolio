Imports System

Module Program
    Sub Main(args As String())
        Dim an, age As Integer
        Dim an_act = 2023
        Console.WriteLine("Entrez votre année de naissance")
        an = Console.ReadLine()
        age = an_act - an
        Select Case age
            Case 0 To 5
                Console.WriteLine("Poussin")
            Case 6 To 10
                Console.WriteLine("Pupille")
            Case 11 To 15
                Console.WriteLine("Cadet")
            Case 16 To 18
                Console.WriteLine("Junior")
            Case 19 To 30
                Console.WriteLine("Senior")
            Case Else
                Console.WriteLine("Vétéran")
        End Select
    End Sub
End Module
