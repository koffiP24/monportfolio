	<script>
		function ajouter(caractere) {
			document.getElementById('resultat').value += caractere;
		}

		function effacer() {
			document.getElementById('resultat').value = '';
		}

		function calculer() {
			var resultat = eval(document.getElementById('resultat').value);
			document.getElementById('resultat').value = resultat;
		}
