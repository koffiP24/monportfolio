#include <stdio.h>
#include <string.h>

int main() 
{
    char mot[100], palindrome[200];
    int longueur, i;

    printf("Entrez un mot : ");
    scanf("%s", &mot);

    longueur = strlen(mot);

    strcpy(palindrome, mot);

    for (i = longueur - 1; i >= 0; i--) 
    {
        palindrome[longueur - 1 - i + longueur] = mot[i];
    }
    palindrome[2 * longueur] = '\0';

    printf("Le palindrome est : %s\n", palindrome);

    // Effacer le contenu du mot d'origine
    memset(mot, 0, sizeof(mot));

    return 0;
}
