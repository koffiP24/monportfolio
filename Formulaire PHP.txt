    <h1>Formulaire de contact</h1>

    <?php
    $nom = $prenom = $email = "";
    $errors = [];
    $success = false;

    if ($_SERVER["REQUEST_METHOD"] == "POST") {
        $nom = trim($_POST["nom"] ?? "");
        $prenom = trim($_POST["prenom"] ?? "");
        $email = trim($_POST["email"] ?? "");

        if (empty($nom)) $errors["nom"] = "Le nom est obligatoire";
        if (empty($prenom)) $errors["prenom"] = "Le prénom est obligatoire";
        if (empty($email)) {
            $errors["email"] = "L'email est obligatoire";
        } elseif (!filter_var($email, FILTER_VALIDATE_EMAIL)) {
            $errors["email"] = "L'email n'est pas valide";
        }

        if (empty($errors)) {
            $fichier = 'contacts.csv';
            
            if (!file_exists($fichier)) {
                $handle = fopen($fichier, 'w');
                fputcsv($handle, ['Nom', 'Prénom', 'Email', 'Date']);
                fclose($handle);
            }
            
            $handle = fopen($fichier, 'a');
            fputcsv($handle, [$nom, $prenom, $email, date('Y-m-d H:i:s')]);
            fclose($handle);
            
            $success = true;
            $nom = $prenom = $email = "";
        }
    }
    ?>

    <?php if ($success): ?>
        <p class="success">Merci ! Vos informations ont été enregistrées avec succès.</p>
    <?php endif; ?>

    <form method="post" action="<?php echo htmlspecialchars($_SERVER["PHP_SELF"]); ?>">
        <div class="form-group">
            <label for="nom">Nom *</label>
            <input type="text" id="nom" name="nom" value="<?php echo htmlspecialchars($nom); ?>">
            <?php if (isset($errors["nom"])): ?>
                <span class="error"><?php echo $errors["nom"]; ?></span>
            <?php endif; ?>
        </div>

        <div class="form-group">
            <label for="prenom">Prénom *</label>
            <input type="text" id="prenom" name="prenom" value="<?php echo htmlspecialchars($prenom); ?>">
            <?php if (isset($errors["prenom"])): ?>
                <span class="error"><?php echo $errors["prenom"]; ?></span>
            <?php endif; ?>
        </div>

        <div class="form-group">
            <label for="email">Email *</label>
            <input type="email" id="email" name="email" value="<?php echo htmlspecialchars($email); ?>">
            <?php if (isset($errors["email"])): ?>
                <span class="error"><?php echo $errors["email"]; ?></span>
            <?php endif; ?>
        </div>

        <div class="form-group">
            <input type="submit" value="Envoyer">
        </div>
    </form>

    <?php
    if (file_exists('contacts.csv')) {
        echo "<h2>Contacts enregistrés</h2>";
        echo "<table border='1' cellpadding='8'>";
        echo "<tr><th>Nom</th><th>Prénom</th><th>Email</th><th>Date</th></tr>";
        
        $file = fopen('contacts.csv', 'r');
        $firstLine = true;
        
        while (($data = fgetcsv($file)) !== false) {
            if ($firstLine) {
                $firstLine = false;
                continue;
            }
            echo "<tr>";
            foreach ($data as $value) {
                echo "<td>".htmlspecialchars($value)."</td>";
            }
            echo "</tr>";
        }
        
        fclose($file);
        echo "</table>";
    }
    ?>
