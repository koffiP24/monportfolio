#include <stdio.h>
#include <math.h>

int main()
{
    double a, b, c, delta, x1, x2;

    printf("Entrez les coefficients de l'equation ax^2 + bx + c = 0 :\n");
    printf("a = ");
    scanf("%lf", &a);
    printf("b = ");
    scanf("%lf", &b);
    printf("c = ");
    scanf("%lf", &c);

    delta = b * b - 4 * a * c;

    if (delta > 0)
    {
        x1 = (-b + sqrt(delta)) / (2 * a);
        x2 = (-b - sqrt(delta)) / (2 * a);
        printf("Les solutions de l'equation sont : x1 = %.2lf et x2 = %.2lf\n", x1, x2);
    }
    else if (delta == 0)
    {
        x1 = -b / (2 * a);
        printf("L'unique solution de l'equation est : x = %.2lf\n", x1);
    }
    else
    {
        printf("L'equation n'a pas de solutions reelles\n");
    }

    return 0;
}
